    display: table;
    width: 100%;
    table-layout: fixed;
    margin-bottom: 20px;
    border-spacing: 5px;
}

div > div {
    display: table-cell;
    border: 1px dotted red;
    padding: 4px 6px;
    width: 100%;
    margin-bottom: 0;
    overflow:hidden;
}
